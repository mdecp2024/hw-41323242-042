def print_star_pyramid(rows):
    for i in range(rows):
        # 打印空格
        print(' ' * (rows - i - 1), end='')
        # 打印星星
        print('*' * (2 * i + 1))

def print_number_pyramid(rows):
    for i in range(rows):
        # 打印空格
        print(' ' * (rows - i - 1), end='')
        # 打印數字
        print(str(i + 1) * (2 * i + 1))
